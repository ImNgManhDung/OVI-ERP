CREATE TABLE ASSET_CATEGORIES (
    ACA_ID              NUMBER NOT NULL,            -- PK
    ASSET_CATEGORY_CODE       VARCHAR2(30 CHAR) NOT NULL, -- Mã loại (VD: TS-MMTB)
    ASSET_CATEGORY_NAME       NVARCHAR2(200) NOT NULL,    -- Tên loại (VD: Máy móc thiết bị)
    
    -- === PHÂN CẤP (HIERARCHY) ===
    PARENT_ID           NUMBER,                     -- Cấu trúc cha con (TS Cố định -> TS Hữu hình -> MMTB)
	LEVEL				NUMBER,						-- Cấp cấu trúc cha con (1,2,3)
    
    -- === PHÂN LOẠI TÀI CHÍNH ===
    -- TANGIBLE (Hữu hình), INTANGIBLE (Vô hình), CIP (XDCB dở dang), PREPAID (Prepaid expenses - Chi phí trả trước)
    ASSET_TYPE          VARCHAR2(20 CHAR) NOT NULL, 
    
    -- === CẤU HÌNH TÀI KHOẢN (GL ACCOUNTS MAPPING) ===
    ASSET_ACC_ID      VARCHAR2(20 CHAR) NOT NULL,     -- Khi ghi tăng, giảm, ghi Nợ/Có TK này (VD: 211, 241)   
    DEPR_ACC_ID      VARCHAR2(20 CHAR) NOT NULL,      -- Khi chạy khấu hao, ghi Có TK này (VD: 214)
    EXPENSE_ACC_ID    VARCHAR2(20 CHAR) NOT NULL,     -- Khi chạy khấu hao, ghi Nợ TK này (VD: 641, 642 - Mặc định)    
    DISPOSAL_LOSS_ACC_ID   VARCHAR2(20 CHAR), 		  -- Khi thanh lý, ghi Nợ TK này (Giá trị còn lại - VD: 811)   
    DISPOSAL_GAIN_ACC_ID   VARCHAR2(20 CHAR),		  -- Khi thanh lý, ghi Có TK này (Doanh thu bán TS - VD: 711)
      
    -- === QUẢN TRỊ ===
    DESCRIPTION             NVARCHAR2(500),
    IS_ACTIVE               VARCHAR2(1 CHAR) DEFAULT 'Y',
    
    -- Audit
    CREATED_BY              VARCHAR2(30 CHAR),
    CREATED_DATE            DATE DEFAULT SYSDATE,
    MODIFIED_BY             VARCHAR2(30 CHAR),
    MODIFY_DATE             DATE,

    CONSTRAINT PK_FA_CATEGORIES PRIMARY KEY (ACA_ID),
    CONSTRAINT UK_FA_CAT_CODE UNIQUE (ASSET_CATEGORY_CODE),
    CONSTRAINT FK_FA_CAT_PARENT FOREIGN KEY (PARENT_ID) REFERENCES ASSET_CATEGORIES (ACA_ID)
);
